258:      * @param array $options
259:      * @return string
260:      **/
261:     public static function tag($tag_name, $content = null, $options = array())
262:     {
263:         $tag = '<'.$tag_name.' '.self::buildAttributes($options);
264: 
265:         if (in_array(strtolower($tag_name), static::$singleTags)
266:             || is_null($content)
267:             ) {
268:             return $tag .= '/>';
269:         } else {
270:             return $tag .= '>'.$content.'</'.$tag_name.'>';
271:         }
272:     }
