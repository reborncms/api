# headers( array $headers )

Set table header field (<th> value)

Set table header field (<th> value)

// Simple Setter
       $ths = array('Title', 'Category', 'Created At');
$table->headers($ths);

// Ok. Use multideminsional array
$ths = array('Name', array('name' => 'Email', 'width' => '25%'), array('name' => 'Created'));
$table->headers($ths);
// Output is...
// <tr>
//   <th>Name</th>
