290:     /**
291:      * Find the asset file in the given path.
292:      * If file is cannot find in given path, return the
293:      * Throw FileNotFoundException at the 'dev' Environment,
294:      * In the other environment (eg: production) return the null.
295:      *
296:      * @param string $type Asset Type (css, js, img)
297:      * @param string $file Asset file name with extension
298:      * @param string|null $module Module name for asset file from module
299:      * @param boolean $noTime Without file modify time
300:      * @return mixed
301:      **/
302:     protected function find($type, $file, $module = null, $noTime = false)
303:     {
304:         if (preg_match('/^http/', $file)) {
305:             return $file;
306:         }
307: 
308:         if (!is_null($module)) {
309:             list($real, $url) = $this->getModuelAssetPath($module);
310: 
311:             $path = $real.$type.DS;
312:             $filePath = $path.$file;
313:             $urlPath = $url.$type.'/'.$file;
314:         } else {
315:             $path = $this->realPath.$type.DS;
316:             $filePath = $path.$file;
317:             $urlPath = $this->path.$type.'/'.$file;
318:         }
319: 
320:         if (file_exists($filePath)) {
321:             if ($noTime) {
322:                 return $urlPath;
323:             }
324: 
325:             $mTime = filemtime($filePath);
326:             return $urlPath.'?'.$mTime;
327:         }
328: 
329: 
330:         // File Not Found Exception Only in Development Stage
331:         $app = Facade::getApplication();
332:         if ($app['env'] == 'dev') {
333:             throw new FileNotFoundException($file, $path);
334:         }
335: 
336:         return null;
337:     }
